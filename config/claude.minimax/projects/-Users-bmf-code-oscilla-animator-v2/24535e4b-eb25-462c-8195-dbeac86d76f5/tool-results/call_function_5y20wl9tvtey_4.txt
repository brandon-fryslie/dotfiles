   513→/**
   514→ * Strided slot write step - writes multiple scalar signal components to contiguous slots.
   515→ *
   516→ * This is the canonical way to materialize multi-component signal values (vec2, vec3, color)
   517→ * into value slots without requiring array-returning evaluators or side-effect kernels.
   518→ *
   519→ * Contract:
   520→ * - inputs.length must equal the stride of slotBase (from slotMeta)
   521→ * - Each input is evaluated as a scalar signal
   522→ * - Results are written sequentially: values.f64[slotBase + i] = evaluateSignal(inputs[i])
   523→ *
   524→ * Example: vec2 output
   525→ *   slotBase = allocSlot(stride=2)
   526→ *   inputs = [sigExprX, sigExprY]
   527→ *   → writes values.f64[slotBase+0] = eval(sigExprX), values.f64[slotBase+1] = eval(sigExprY)
   528→ */
   529→export interface StepSlotWriteStrided {
   530→  readonly kind: 'slotWriteStrided';
   531→  readonly slotBase: ValueSlot;
   532→  readonly inputs: readonly SigExprId[];
   533→}
