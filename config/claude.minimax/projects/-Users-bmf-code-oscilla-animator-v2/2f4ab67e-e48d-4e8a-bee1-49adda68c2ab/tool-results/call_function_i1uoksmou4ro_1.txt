   285→  fieldId: FieldExprId,
   286→  fieldExprs: readonly FieldExpr[]
   287→): InstanceId | undefined {
   288→  const expr = fieldExprs[fieldId as number];
   289→  if (!expr) return undefined;
   290→
   291→  switch (expr.kind) {
   292→    case 'intrinsic':
   293→    case 'array':
   294→    case 'stateRead':
   295→      return expr.instanceId;
   296→    case 'map':
   297→      return expr.instanceId ?? inferFieldInstanceFromExprs(expr.input, fieldExprs);
   298→    case 'zip':
   299→      // Take first input's instance (they should all match per validation)
   300→      return expr.instanceId ?? (expr.inputs.length > 0 ? inferFieldInstanceFromExprs(expr.inputs[0], fieldExprs) : undefined);
   301→    case 'zipSig':
   302→      return expr.instanceId ?? inferFieldInstanceFromExprs(expr.field, fieldExprs);
   303→    case 'broadcast':
   304→    case 'const':
   305→      return undefined;
   306→    default:
   307→      return undefined;
   308→  }
